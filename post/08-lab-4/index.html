from keras.models import Sequential

Modeli oluştur

model = Sequential()
model.add(Embedding(max_features, maxlen))
model.add(Dropout(0.2))
model.add(LSTM(128))
model.add(Dropout(0.2))
model.add(Dense(1, activation='sigmoid'))

model.compile(optimizer='rmsprop', loss='binary_crossentropy', metrics=['acc'])
model.summary()
